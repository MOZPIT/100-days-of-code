Day 6 of 100 Days of Code


Will you pass?

Description:Build a program in JS that tells users if they pass a university entrance exam. There are 5 modules and to pass the entrance exam, a user must past 4 modules with an average grade of 60.

University Entrance Exam:

JS Code:

            
                /*jslint devel: true */
                let grades = []
                var gradeSum = 0
                var passModule = 0
                var average = 0
                var results = "This is what you have scored today:\n\n

"

                for(var i = 1; i <= 5; i++){
                    grades[i] = prompt("Enter Grade For Module " + i)
                    gradeSum += parseInt(grades[i])
                    if(grades[i] >= 60){
                        passModule++
                    }
                    results += "module " + i + ": " + grades[i] + ".\n\n

"
                }
                average = (gradeSum/5).toFixed(2)

                if(average >= 60 && passModule >= 4){
                    results += "You passed " + passModule + " modules\n
"
                    results += "Your average is " + average + "\n
"
                    results += "Congratulations you have passed!"       
                }else{
                    results += "You have failed " + (5-passModule) + " module(s)\n
"
                    results += "Your average is " + average + "\n
"
                    results += "Wish you better luck next time :-("
                }

                document.getElementById("result").innerHTML = results
                console.log(results)
